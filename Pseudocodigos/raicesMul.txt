FUNCTION raicesMul(f,f1,f2,x0,tolerance,nMax)
	xi <- x0
	fxi <- f(xi)

		PRINT('A root has been found and it is: '+ xi)
	ELSE

		f1xi <- f1(xi)
		f2xi <- f2(xi)
		error <- tolerance + 1
		det <- (f1xi^2)-(fxi*f2xi)
	
		WHILE fxi != 0 AND error > tolerance AND counter < nMax DO
			xiAux <- xi
			xi <- xi - ((fxi*f1xi)/((f1xi^2)-(fxi*f2xi)))

			f1xi <- f1(xi)
			f2xi <- f2(xi)

			det <- (f1xi^2)-(fxi*f2xi)
			counter <- counter +1
		END WHILE	

		IF fxi == 0

        ELSE IF error <= tolerance
            PRINT('One approach has been found and it is: '+ xi)
        ELSE IF det == 0
            PRINT('Error during method execution')
        ELSE 
            PRINT('The method fails with the maximum number of iterations given')
        END IF

	END IF
END FUNCTION
